def test_decision_fully_replayable(conn):
    """Per Apr26 F-011 acceptance: every edge decision is replayable.
    Given a venue_commands row + linked snapshot + envelope, can we reconstruct?"""
    cmd = get_command_by_id(conn, 'C1')
    snap = get_snapshot_by_id(conn, cmd.snapshot_fk)
    env = get_envelope_by_command_id(conn, cmd.command_id)
    # All forensic fields present:
    assert snap.raw_clob_snapshot_jsonb is not None
    assert snap.raw_orderbook_jsonb is not None
    assert cmd.signed_order_blob is not None
    assert cmd.post_order_response_blob is not None
    assert cmd.condition_id is not None
    assert cmd.outcome in ('YES', 'NO')
    assert env.raw_envelope_jsonb is not None
    # Replay: reconstruct decision from raw bytes alone
    replayed = replay_decision_from_raw(snap, cmd, env)
    assert replayed.token_id == cmd.token_id
    assert replayed.price == cmd.price
